18const AGE_MAGIC: &[u8] = b"age-encryption.org/";
19const V1_MAGIC: &[u8] = b"v1";
20const MAC_TAG: &[u8] = b"---";
21const ENCODED_MAC_LENGTH: usize = 43;
22
23#[derive(Debug, PartialEq)]
24pub(crate) struct HeaderV1 {
25    pub(crate) recipients: Vec<Stanza>,
26    pub(crate) mac: [u8; 32],
27    /// The serialized bytes of this header. Set if we parsed this header from a reader,
28    /// to handle the possibility that the header is not round-trip canonical, such as it
29    /// containing a legacy stanza with a body of length 0 mod 64.
30    ///
31    /// We do not write this back out in `Header::write`, because we never write headers
32    /// that we have parsed, and headers we generate for writing will never have this set.
33    encoded_bytes: Option<Vec<u8>>,
34}
35
36impl HeaderV1 {
37    pub(crate) fn new(recipients: Vec<Stanza>, mac_key: HmacKey) -> Result<Self, EncryptError> {
38        let mut header = HeaderV1 {
39            recipients,
40            mac: [0; 32],
41            encoded_bytes: None,
42        };
43
44        if header.no_scrypt() {
45            // Keep the joint well oiled!
46            header.recipients.push(grease_the_joint());
47        }
48
49        if !header.is_valid() {
50            return Err(EncryptError::MixedRecipientAndPassphrase);
51        }
52
53        let mut mac = HmacWriter::new(mac_key);
54        cookie_factory::gen(write::header_v1_minus_mac(&header), &mut mac)
55            .expect("can serialize Header into HmacWriter");
56        header
57            .mac
58            .copy_from_slice(mac.finalize().into_bytes().as_slice());
59
60        Ok(header)
61    }
62
63    pub(crate) fn verify_mac(&self, mac_key: HmacKey) -> Result<(), hmac::digest::MacError> {
64        let mut mac = HmacWriter::new(mac_key);
65        if let Some(bytes) = &self.encoded_bytes {
66            // The MAC covers all bytes up to the end of the --- prefix.
67            mac.write_all(&bytes[..bytes.len() - ENCODED_MAC_LENGTH - 2])
68                .expect("can serialize Header into HmacWriter");
69        } else {
70            cookie_factory::gen(write::header_v1_minus_mac(self), &mut mac)
71                .expect("can serialize Header into HmacWriter");
72        }
73        mac.verify(&self.mac)
74    }
75
76    fn any_scrypt(&self) -> bool {
77        self.recipients
78            .iter()
79            .any(|r| r.tag == scrypt::SCRYPT_RECIPIENT_TAG)
80    }
81
82    /// Checks whether the header contains a single recipient of type `scrypt`.
83    ///
84    /// This can be used along with [`Self::no_scrypt`] to enforce the structural
85    /// requirements on the v1 header.
86    pub(crate) fn valid_scrypt(&self) -> bool {
87        self.any_scrypt() && self.recipients.len() == 1
88    }
89
90    /// Checks whether the header contains no `scrypt` recipients.
91    ///
92    /// This can be used along with [`Self::valid_scrypt`] to enforce the structural
93    /// requirements on the v1 header.
94    pub(crate) fn no_scrypt(&self) -> bool {
95        !self.any_scrypt()
96    }
97
98    /// Enforces structural requirements on the v1 header.
99    pub(crate) fn is_valid(&self) -> bool {
100        self.valid_scrypt() || self.no_scrypt()
101    }
102}

104impl Header {
105    pub(crate) fn read<R: Read>(mut input: R) -> Result<Self, DecryptError> {
106        let mut data = vec![];
107        loop {
108            match read::header(&data) {
109                Ok((_, mut header)) => {
110                    if let Header::V1(h) = &mut header {
111                        h.encoded_bytes = Some(data);
112                    }
113                    break Ok(header);
114                }
115                Err(nom::Err::Incomplete(nom::Needed::Size(n))) => {
116                    // Read the needed additional bytes. We need to be careful how the
117                    // parser is constructed, because if we read more than we need, the
118                    // remainder of the input will be truncated.
119                    let m = data.len();
120                    let new_len = m + n.get();
121                    data.resize(new_len, 0);
122                    input.read_exact(&mut data[m..new_len])?;
123                }
124                Err(_) => {
125                    break Err(DecryptError::InvalidHeader);
126                }
127            }
128        }
129    }
130
131    pub(crate) fn read_buffered<R: BufRead>(mut input: R) -> Result<Self, DecryptError> {
132        let mut data = vec![];
133        loop {
134            match read::header(&data) {
135                Ok((_, mut header)) => {
136                    if let Header::V1(h) = &mut header {
137                        h.encoded_bytes = Some(data);
138                    }
139                    break Ok(header);
140                }
141                Err(nom::Err::Incomplete(nom::Needed::Size(_))) => {
142                    // As we have a buffered reader, we can leverage the fact that the
143                    // currently-defined header formats are newline-separated, to more
144                    // efficiently read data for the parser to consume.
145                    if input.read_until(b'\n', &mut data)? == 0 {
146                        break Err(DecryptError::Io(io::Error::new(
147                            io::ErrorKind::UnexpectedEof,
148                            "Incomplete header",
149                        )));
150                    }
151                }
152                Err(_) => {
153                    break Err(DecryptError::InvalidHeader);
154                }
155            }
156        }
157    }

221    pub(crate) fn write<W: Write>(&self, mut output: W) -> io::Result<()> {
222        cookie_factory::gen(write::header(self), &mut output)
223            .map(|_| ())
224            .map_err(|e| {
225                io::Error::new(
226                    io::ErrorKind::Other,
227                    format!("failed to write header: {}", e),
228                )
229            })
230    }

246    }

255mod read {
256    use age_core::format::read::{arbitrary_string, legacy_age_stanza};
257    use nom::{
258        branch::alt,
259        bytes::streaming::{tag, take},
260        character::streaming::newline,
261        combinator::{map, map_opt},
262        multi::many1,
263        sequence::{pair, preceded, terminated},
264        IResult,
265    };
266
267    use super::*;
268    use crate::util::read::base64_arg;
269
270    fn header_v1(input: &[u8]) -> IResult<&[u8], HeaderV1> {
271        preceded(
272            pair(tag(V1_MAGIC), newline),
273            map(
274                pair(
275                    many1(legacy_age_stanza),
276                    preceded(
277                        pair(tag(MAC_TAG), tag(b" ")),
278                        terminated(
279                            map_opt(take(ENCODED_MAC_LENGTH), |tag| {
280                                base64_arg::<_, 32, 33>(&tag)
281                            }),
282                            newline,
283                        ),
284                    ),
285                ),
286                |(recipients, mac)| HeaderV1 {
287                    recipients: recipients.into_iter().map(Stanza::from).collect(),
288                    mac,
289                    encoded_bytes: None,
290                },
291            ),
292        )(input)
293    }
294
295    /// From the age specification:
296    /// ```text
297    /// The first line of the header is age-encryption.org/ followed by an arbitrary
298    /// version string. ... We describe version v1, other versions can change anything
299    /// after the first line.
300    /// ```
301    pub(super) fn header(input: &[u8]) -> IResult<&[u8], Header> {
302        preceded(
303            tag(AGE_MAGIC),
304            alt((
305                map(header_v1, Header::V1),
306                map(terminated(arbitrary_string, newline), |s| {
307                    Header::Unknown(s.to_string())
308                }),
309            )),
310        )(input)
311    }
312}

314mod write {
315    use age_core::format::write::age_stanza;
316    use cookie_factory::{
317        combinator::{slice, string},
318        multi::all,
319        sequence::tuple,
320        SerializeFn, WriteContext,
321    };
322    use std::io::Write;
323
324    use super::*;
325    use crate::util::write::encoded_data;
326
327    fn recipient_stanza<'a, W: 'a + Write>(r: &'a Stanza) -> impl SerializeFn<W> + 'a {
328        move |w: WriteContext<W>| {
329            let args: Vec<_> = r.args.iter().map(|s| s.as_str()).collect();
330            let writer = age_stanza(&r.tag, &args, &r.body);
331            writer(w)
332        }
333    }
334
335    pub(super) fn header_v1_minus_mac<'a, W: 'a + Write>(
336        h: &'a HeaderV1,
337    ) -> impl SerializeFn<W> + 'a {
338        tuple((
339            slice(AGE_MAGIC),
340            slice(V1_MAGIC),
341            string("\n"),
342            all(h.recipients.iter().map(move |r| recipient_stanza(r))),
343            slice(MAC_TAG),
344        ))
345    }
346
347    fn header_v1<'a, W: 'a + Write>(h: &'a HeaderV1) -> impl SerializeFn<W> + 'a {
348        tuple((
349            header_v1_minus_mac(h),
350            string(" "),
351            encoded_data(&h.mac),
352            string("\n"),
353        ))
354    }
355
356    pub(super) fn header<'a, W: 'a + Write>(h: &'a Header) -> impl SerializeFn<W> + 'a {
357        move |w: WriteContext<W>| match h {
358            Header::V1(v1) => header_v1(v1)(w),
359            Header::Unknown(version) => tuple((slice(AGE_MAGIC), slice(version), string("\n")))(w),
360        }
361    }
