20pub(crate) struct Nonce([u8; 16]);
21
22impl AsRef<[u8]> for Nonce {
23    fn as_ref(&self) -> &[u8] {
24        &self.0
25    }
26}
27
28impl Nonce {
29    fn random() -> Self {
30        let mut nonce = [0; 16];
31        OsRng.fill_bytes(&mut nonce);
32        Nonce(nonce)
33    }
34
35    fn read<R: Read>(input: &mut R) -> io::Result<Self> {
36        let mut nonce = [0; 16];
37        input.read_exact(&mut nonce)?;
38        Ok(Nonce(nonce))
39    }
40
41    #[cfg(feature = "async")]
42    #[cfg_attr(docsrs, doc(cfg(feature = "async")))]
43    async fn read_async<R: AsyncRead + Unpin>(input: &mut R) -> io::Result<Self> {
44        let mut nonce = [0; 16];
45        input.read_exact(&mut nonce).await?;
46        Ok(Nonce(nonce))
47    }
48}

50/// Encryptor for creating an age file.
51pub struct Encryptor {
52    header: Header,
53    nonce: Nonce,
54    payload_key: PayloadKey,
55}
56
57impl Encryptor {
58    /// Returns an `Encryptor` that will create an age file encrypted with a passphrase.
59    /// Anyone with the passphrase can decrypt the file.
60    ///
61    /// This API should only be used with a passphrase that was provided by (or generated
62    /// for) a human. For programmatic use cases, instead generate an [`x25519::Identity`]
63    /// and then use [`Encryptor::with_recipients`].
64    ///
65    /// [`x25519::Identity`]: crate::x25519::Identity
66    pub fn with_user_passphrase(passphrase: SecretString) -> Self {
67        Self::with_recipients(iter::once(&scrypt::Recipient::new(passphrase) as _))
68            .expect("no errors can occur with this recipient set")
69    }
70
71    /// Constructs an `Encryptor` that will create an age file encrypted to a list of
72    /// recipients.
73    pub fn with_recipients<'a>(
74        recipients: impl Iterator<Item = &'a dyn Recipient>,
75    ) -> Result<Self, EncryptError> {
76        let file_key = new_file_key();
77
78        let recipients = {
79            let mut control = None;
80
81            let mut stanzas = vec![];
82            let mut have_recipients = false;
83            for recipient in recipients {
84                have_recipients = true;
85                let (mut r_stanzas, r_labels) = recipient.wrap_file_key(&file_key)?;
86
87                if let Some(l_labels) = control.take() {
88                    if l_labels != r_labels {
89                        // Improve error message.
90                        let err = if stanzas
91                            .iter()
92                            .chain(&r_stanzas)
93                            .any(|stanza| stanza.tag == crate::scrypt::SCRYPT_RECIPIENT_TAG)
94                        {
95                            EncryptError::MixedRecipientAndPassphrase
96                        } else {
97                            EncryptError::IncompatibleRecipients { l_labels, r_labels }
98                        };
99                        return Err(err);
100                    }
101                    control = Some(l_labels);
102                } else if r_labels.iter().all(is_arbitrary_string) {
103                    control = Some(r_labels);
104                } else {
105                    return Err(EncryptError::InvalidRecipientLabels(r_labels));
106                }
107
108                stanzas.append(&mut r_stanzas);
109            }
110            if !have_recipients {
111                return Err(EncryptError::MissingRecipients);
112            }
113            stanzas
114        };
115
116        let header = HeaderV1::new(recipients, mac_key(&file_key))?;
117        let nonce = Nonce::random();
118        let payload_key = v1_payload_key(&file_key, &header, &nonce).expect("MAC is correct");
119
120        Ok(Self {
121            header: Header::V1(header),
122            nonce,
123            payload_key,
124        })
125    }
126
127    /// Creates a wrapper around a writer that will encrypt its input.
128    ///
129    /// Returns errors from the underlying writer while writing the header.
130    ///
131    /// You **MUST** call [`StreamWriter::finish`] when you are done writing, in order to
132    /// finish the encryption process. Failing to call [`StreamWriter::finish`] will
133    /// result in a truncated file that will fail to decrypt.
134    pub fn wrap_output<W: Write>(self, mut output: W) -> io::Result<StreamWriter<W>> {
135        let Self {
136            header,
137            nonce,
138            payload_key,
139        } = self;
140        header.write(&mut output)?;
141        output.write_all(nonce.as_ref())?;
142        Ok(Stream::encrypt(payload_key, output))
143    }
144
145    /// Creates a wrapper around a writer that will encrypt its input.
146    ///
147    /// Returns errors from the underlying writer while writing the header.
148    ///
149    /// You **MUST** call [`AsyncWrite::poll_close`] when you are done writing, in order
150    /// to finish the encryption process. Failing to call [`AsyncWrite::poll_close`]
151    /// will result in a truncated file that will fail to decrypt.
152    #[cfg(feature = "async")]
153    #[cfg_attr(docsrs, doc(cfg(feature = "async")))]
154    pub async fn wrap_async_output<W: AsyncWrite + Unpin>(
155        self,
156        mut output: W,
157    ) -> io::Result<StreamWriter<W>> {
158        let Self {
159            header,
160            nonce,
161            payload_key,
162        } = self;
163        header.write_async(&mut output).await?;
164        output.write_all(nonce.as_ref()).await?;
165        Ok(Stream::encrypt_async(payload_key, output))
166    }
167}

169/// Decryptor for an age file.
170pub struct Decryptor<R> {
171    /// The age file.
172    input: R,
173    /// The age file's header.
174    header: Header,
175    /// The age file's AEAD nonce
176    nonce: Nonce,
177}
178
179impl<R> Decryptor<R> {
180    fn from_v1_header(input: R, header: HeaderV1, nonce: Nonce) -> Result<Self, DecryptError> {
181        // Enforce structural requirements on the v1 header.
182        if header.is_valid() {
183            Ok(Self {
184                input,
185                header: Header::V1(header),
186                nonce,
187            })
188        } else {
189            Err(DecryptError::InvalidHeader)
190        }
191    }
192
193    /// Returns `true` if the age file is encrypted to a passphrase.
194    pub fn is_scrypt(&self) -> bool {
195        match &self.header {
196            Header::V1(header) => header.valid_scrypt(),
197            Header::Unknown(_) => false,
198        }
199    }
200
201    fn obtain_payload_key<'a>(
202        &self,
203        mut identities: impl Iterator<Item = &'a dyn Identity>,
204    ) -> Result<PayloadKey, DecryptError> {
205        match &self.header {
206            Header::V1(header) => identities
207                .find_map(|key| key.unwrap_stanzas(&header.recipients))
208                .unwrap_or(Err(DecryptError::NoMatchingKeys))
209                .and_then(|file_key| v1_payload_key(&file_key, header, &self.nonce)),
210            Header::Unknown(_) => unreachable!(),
211        }
212    }
213}

215impl<R: Read> Decryptor<R> {
216    /// Attempts to create a decryptor for an age file.
217    ///
218    /// Returns an error if the input does not contain a valid age file.
219    ///
220    /// # Performance
221    ///
222    /// This constructor will work with any type implementing [`io::Read`], and uses a
223    /// slower parser and internal buffering to ensure no overreading occurs. Consider
224    /// using [`Decryptor::new_buffered`] for types implementing `std::io::BufRead`, which
225    /// includes `&[u8]` slices.
226    pub fn new(mut input: R) -> Result<Self, DecryptError> {
227        let header = Header::read(&mut input)?;
228
229        match header {
230            Header::V1(v1_header) => {
231                let nonce = Nonce::read(&mut input)?;
232                Decryptor::from_v1_header(input, v1_header, nonce)
233            }
234            Header::Unknown(_) => Err(DecryptError::UnknownFormat),
235        }
236    }
237
238    /// Attempts to decrypt the age file.
239    ///
240    /// If successful, returns a reader that will provide the plaintext.
241    pub fn decrypt<'a>(
242        self,
243        identities: impl Iterator<Item = &'a dyn Identity>,
244    ) -> Result<StreamReader<R>, DecryptError> {
245        self.obtain_payload_key(identities)
246            .map(|payload_key| Stream::decrypt(payload_key, self.input))
247    }
248}

356    fn recipient_round_trip<'a>(
357        recipients: impl Iterator<Item = &'a dyn Recipient>,
358        identities: impl Iterator<Item = &'a dyn Identity>,
359    ) {
360        let test_msg = b"This is a test message. For testing.";
361
362        let mut encrypted = vec![];
363        let e = Encryptor::with_recipients(recipients).unwrap();
364        {
365            let mut w = e.wrap_output(&mut encrypted).unwrap();
366            w.write_all(test_msg).unwrap();
367            w.finish().unwrap();
368        }
369
370        let d = Decryptor::new(&encrypted[..]).unwrap();
371        let mut r = d.decrypt(identities).unwrap();
372        let mut decrypted = vec![];
373        r.read_to_end(&mut decrypted).unwrap();
374
375        assert_eq!(&decrypted[..], &test_msg[..]);
376    }

378    #[cfg(feature = "async")]
379    fn recipient_async_round_trip<'a>(
380        recipients: impl Iterator<Item = &'a dyn Recipient>,
381        identities: impl Iterator<Item = &'a dyn Identity>,
382    ) {
383        let test_msg = b"This is a test message. For testing.";
384        let mut cx = noop_context();
385
386        let mut encrypted = vec![];
387        let e = Encryptor::with_recipients(recipients).unwrap();
388        {
389            let w = {
390                let f = e.wrap_async_output(&mut encrypted);
391                pin_mut!(f);
392
393                loop {
394                    match f.as_mut().poll(&mut cx) {
395                        Poll::Ready(Ok(w)) => break w,
396                        Poll::Ready(Err(e)) => panic!("Unexpected error: {}", e),
397                        Poll::Pending => panic!("Unexpected Pending"),
398                    }
399                }
400            };
401            pin_mut!(w);
402
403            let mut tmp = &test_msg[..];
404            loop {
405                match w.as_mut().poll_write(&mut cx, tmp) {
406                    Poll::Ready(Ok(0)) => break,
407                    Poll::Ready(Ok(written)) => tmp = &tmp[written..],
408                    Poll::Ready(Err(e)) => panic!("Unexpected error: {}", e),
409                    Poll::Pending => panic!("Unexpected Pending"),
410                }
411            }
412            loop {
413                match w.as_mut().poll_close(&mut cx) {
414                    Poll::Ready(Ok(())) => break,
415                    Poll::Ready(Err(e)) => panic!("Unexpected error: {}", e),
416                    Poll::Pending => panic!("Unexpected Pending"),
417                }
418            }
419        }
420
421        let d = {
422            let f = Decryptor::new_async(&encrypted[..]);
423            pin_mut!(f);
424
425            loop {
426                match f.as_mut().poll(&mut cx) {
427                    Poll::Ready(Ok(w)) => break w,
428                    Poll::Ready(Err(e)) => panic!("Unexpected error: {}", e),
429                    Poll::Pending => panic!("Unexpected Pending"),
430                }
431            }
432        };
433
434        let decrypted = {
435            let mut buf = vec![];
436            let r = d.decrypt_async(identities).unwrap();
437            pin_mut!(r);
438
439            let mut tmp = [0; 4096];
440            loop {
441                match r.as_mut().poll_read(&mut cx, &mut tmp) {
442                    Poll::Ready(Ok(0)) => break buf,
443                    Poll::Ready(Ok(read)) => buf.extend_from_slice(&tmp[..read]),
444                    Poll::Ready(Err(e)) => panic!("Unexpected error: {}", e),
445                    Poll::Pending => panic!("Unexpected Pending"),
446                }
447            }
448        };
449
450        assert_eq!(&decrypted[..], &test_msg[..]);
451    }
