20/// `HMAC[key](message)`
21///
22/// HMAC from [RFC 2104] with SHA-256.
23///
24/// [RFC 2104]: https://tools.ietf.org/html/rfc2104
25pub(crate) struct HmacWriter {
26    inner: Hmac<Sha256>,
27}
28
29impl HmacWriter {
30    /// Constructs a new writer to process input data.
31    pub(crate) fn new(key: HmacKey) -> Self {
32        HmacWriter {
33            inner: Hmac::new_from_slice(key.0.expose_secret()).expect("key is the correct length"),
34        }
35    }
36
37    /// Checks if `mac` is correct for the processed input.
38    pub(crate) fn finalize(self) -> CtOutput<Hmac<Sha256>> {
39        self.inner.finalize()
40    }
41
42    /// Checks if `mac` is correct for the processed input.
43    pub(crate) fn verify(self, mac: &[u8]) -> Result<(), MacError> {
44        self.inner.verify_slice(mac)
45    }
46}
47
48impl Write for HmacWriter {
49    fn write(&mut self, data: &[u8]) -> io::Result<usize> {
50        self.inner.update(data);
51        Ok(data.len())
52    }
53
54    fn flush(&mut self) -> io::Result<()> {
55        Ok(())
56    }
57}

59/// `scrypt[salt, N](password)`
60///
61/// scrypt from [RFC 7914] with r = 8 and P = 1. N must be a power of 2.
62///
63/// [RFC 7914]: https://tools.ietf.org/html/rfc7914
64pub(crate) fn scrypt(salt: &[u8], log_n: u8, password: &str) -> Result<[u8; 32], InvalidParams> {
65    let params = ScryptParams::new(log_n, 8, 1, 32)?;
66
67    let mut output = [0; 32];
68    scrypt_inner(password.as_bytes(), salt, &params, &mut output)
69        .expect("output is the correct length");
70    Ok(output)
71}
